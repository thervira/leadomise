[docs]def generateFragmentLibrary(mols, minFragmentSize=1):
    """Generates set of fragments from the input file.

    :param mols: rdkit sdf supplier
        input ligands
    :param minFragmentSize: int (optional)
        minimum size of the fragment (default is 1)

    :returns fragmentLibrary: set
        set of fragments generated from ligands from input file
    """

    #allMols = Chem.SDMolSupplier(in_file)

    fragmentLibrary = set()
    for mol in mols:
        frs = BRICS.BRICSDecompose(mol, minFragmentSize=minFragmentSize, returnMols=True)
        fragmentLibrary.update(frs)

    return fragmentLibrary
